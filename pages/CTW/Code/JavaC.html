Java and C#

Project Description

This project was a final test of my proficiency in the Java programming language and my ability to quickly adapt to a new language (in this case C#). The goal was to create an ATM program with deposit and withdraw functions while keeping track of the machine balance in each type bill. The program was written in both Java and C#, both of which had the same console output and usable commands.

C# class written to deal with user input

 


using System;
namespace ConsoleATM
{
    class input
    {
        static void Main(string[] args)
        {
            String input;
            ATM machine = new ATM();
            bool run = true;
            bool  valid = false;
            while (run)
            {
                input = Console.ReadLine();
                if (input[0] == 'R') 
                {
                    machine.restock();
                    valid = true;
                }
                if (input[0] == 'W')
                {
                    String[] value = input.Split("$");
                    if (value.Length > 1)
                    {
                        machine.withdrawl(Int32.Parse(value[1]));
                        valid = true;
                    }
                    else
                    {
                        Console.WriteLine("Failure: Invalid Command");
                        Console.ReadLine();
                    }
                }
                if (input[0] == 'I') 
                {
                    machine.inquiry(input);
                    valid = true;
                }
                if ((input[0]) == 'Q') 
                {
                    run = false;
                    valid = true;
                }
                if (!valid)
                {
                    Console.WriteLine("Invalid Command");
                }
                else
                {
                    valid = false;
                }
            }
        }
    }
}

using System;
    using System.Collections.Generic;
    using System.Text;

    namespace ConsoleATM
    {
        class ATM
        {
            private int hundred;
            private int fifty;
            private int twenty;
            private int ten;
            private int five;
            private int one;

            public ATM()
            {
                hundred = 10;
                fifty = 10;
                twenty = 10;
                ten = 10;
                five = 10;
                one = 10;
            }

            public void restock()
            {
                hundred = 10;
                fifty = 10;
                twenty = 10;
                ten = 10;
                five = 10;
                one = 10;
                this.readBalance();
            }

            public void withdrawl(int dollars)
            {
                int total = dollars;
                bool funds = true;
                int outHundred = 0;
                int outFifty = 0;
                int outTwenty = 0;
                int outTen = 0;
                int outFive = 0;
                int outOne = 0;

                if (total / 100 >= 1)
                {
                    outHundred = total / 100;
                    if (outHundred > hundred)
                    {
                        funds = false;
                    }
                    total = (total % 100);
                }
                if (total / 50 >= 1)
                {
                    outFifty = total / 50;
                    if (outFifty > fifty)
                    {
                        funds = false;
                    }
                    total = (total % 50);
                }
                if (total / 20 >= 1)
                {
                    outTwenty = total / 20;
                    if (outTwenty > twenty)
                    {
                        funds = false;
                    }
                    total = (total % 20);
                }
                if (total / 10 >= 1)
                {
                    outTen = total / 10;
                    if (outTen > ten)
                    {
                        funds = false;
                    }
                    total = (total % 10);
                }
                if (total / 5 >= 1)
                {
                    outFive = total / 5;
                    if (outFive > five)
                    {
                        funds = false;
                    }
                    total = (total % 5);
                }
                if (total >= 1)
                {
                    outOne = total;
                    if (outOne > one)
                    {
                        funds = false;
                    }
                }
                if (funds)
                {
                    hundred = hundred - outHundred;
                    fifty = fifty - outFifty;
                    twenty = twenty - outTwenty;
                    ten = ten - outTen;
                    five = five - outFive;
                    one = one - outOne;
                    Console.WriteLine("Success: dispenced $" + dollars);
                    this.readBalance();
                }
                else
                {
                    Console.WriteLine("Failure: insufficient funds");
                    this.readBalance();
                }
            }

            public void readBalance()
            {
                Console.WriteLine("Machine Balance:");
                Console.WriteLine("$100 - " + hundred);
                Console.WriteLine("$50 - " + fifty);
                Console.WriteLine("$20 - " + twenty);
                Console.WriteLine("$10 - " + ten);
                Console.WriteLine("$5 - " + five);
                Console.WriteLine("$1 - " + one);
            }

            public void inquiry(String denominations)
            {
                String[] values = denominations.Split("$");
                if (values.Length > 1)
                {
                    Console.WriteLine(values[1]);
                    bool error = false;
                    bool match = false;
                    for (int i = 1; i < values.Length; i++)
                    {
                        match = false;
                        if (values[i].Equals("100"))
                        {
                            match = true;
                            Console.WriteLine("$100 - " + hundred);
                        }
                        if (values[i].Equals("50"))
                        {
                            match = true;
                            Console.WriteLine("$50 - " + fifty);
                        }
                        if (values[i].Equals("20"))
                        {
                            match = true;
                            Console.WriteLine("$20 - " + twenty);
                        }
                        if (values[i].Equals("10"))
                        {
                            match = true;
                            Console.WriteLine("$10 - " + ten);
                        }
                        if (values[i].Equals("5"))
                        {
                            match = true;
                            Console.WriteLine("$5 - " + five);
                        }
                        if (values[i].Equals("1"))
                        {
                            match = true;
                            Console.WriteLine("$1 - " + one);
                        }
                        if (match == false)
                        {
                            Console.WriteLine("Failure: invalid command");
                        }
                    }
                }
                else
                {
                    Console.WriteLine("Failure: invalid command");
                }
            }
        }
    }

            {

                    valid=true;
                }
                if (in.charAt(0) == 'W') 
                {
                    String[] value = in.split("\\$");
                    if (value.length>1)
                    {
                        machine.withdrawl(Integer.parseInt(value[1]));
                        valid=true;
                    }
                    else
                    {
                        System.out.println("Failure: Invalid Command");
                        input.nextLine();
                    }
                }
                if (in.charAt(0) == 'I')
                {
                    machine.inquiry(in);
                    valid=true;
                }
                if (in.charAt(0) == 'Q')
                {
                    run = false;
                    valid=true;
                }
                if (valid==false)
                {
                    System.out.println("Invalid Command");
                }
                else
                {
                    valid=false;
                }
            }
        }
    }

Project Reflection

While every piece of this project is fairly similar to its counterpart (Java to C#, and Eclipse to Visual Studio) I was still pushed out of my comfort zone and learned a lot, even becoming fairly comfortable with these new tools. I found that if I was able to conceptualize the algorithms first without thinking about syntax, I was able to more efficiently and effectively write the program in both languages. Overall, from this project I learned how to use Visual Studio and how to code in C#, while becoming much more comfortable with Source Control as well as the agile development process.
